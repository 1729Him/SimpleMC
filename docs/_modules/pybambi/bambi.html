[docs]def run_pyBAMBI(loglikelihood, prior, nDims, **kwargs):
    """Run pyBAMBI.

    Parameters
    ----------
    sampler: str
        Choice of sampler. Options: `['mnest', 'snest']`.
        Default `'multinest'`.

    nlive: int
        Number of live points.
        Default `nDims*25`

    root: str
        root of filename.
        Default `'chains/<nested_sampler>'`

    num_repeats: int
        number of repeats for polychord.
        Default `nDims*5`

    eff: float
        efficiency for multinest.
        Default `0.5**nDims`

    learner: object
        information indicating what learning algorithm to use for approximating
        the likelihood. Can be the string `'keras'`, or a `keras.models.Model`
        Default `'keras'`

    ntrain: int
        Number of training points to use
        Default `nlive`

    proxy_tolerance: float
        Required accuracy of proxy.
        Default `0.01`

    ns_output: int
        Nested sampling output level.

    """
    # Process kwargs
    nestedType = kwargs.pop('nestedType', 'multi')
    dynamic = kwargs.pop('dynamic', 'no')
    nlive = kwargs.pop('nlive', nDims*25)
    root = kwargs.pop('root', os.path.join('chains', nestedType, '/'))
    num_repeats = kwargs.pop('num_repeats', nDims*5)
    eff = kwargs.pop('eff', 0.5**nDims)
    learner = kwargs.pop('learner', 'keras')
    proxy_tolerance = kwargs.pop('proxy_tolerance', 0.1)
    failure_tolerance = kwargs.pop('failure_tolerance', 0.5)
    ntrain = kwargs.pop('ntrain', nlive)
    nprocess = kwargs.pop('nprocess', 0)

    if kwargs:
        raise TypeError('Unexpected **kwargs: %r' % kwargs)

    # Set up the global manager of the BAMBI session.
    thumper = BambiManager(loglikelihood, learner, proxy_tolerance,
                           failure_tolerance, ntrain)    
        
    if nprocess == 0: 
        ncores = mp.cpu_count()    
        nprocess = ncores//2

    pool = mp.Pool(processes=nprocess)

    try:
        if dynamic == 'yes':
            print("Using dynamic nested sampling...")
            sampler = dynesty.DynamicNestedSampler(thumper.loglikelihood, prior, ndim=nDims,\
                            bound = nestedType, sample='rwalk',\
                            pool = pool, queue_size = nprocess)
            
            sampler.run_nested(nlive)
      
        else:
            sampler = dynesty.NestedSampler(thumper.loglikelihood, prior, ndim=nDims,\
                                bound = nestedType, sample = 'unif', nlive = nlive,\
                                pool = pool, queue_size = nprocess)

            sampler.run_nested(dlogz=0.5)
    except:
        raise NotImplementedError('sampler %s is not implemented' % nestedType)
        sys.exit()
    
    M = sampler.results
    weights = np.exp(M['logwt'] - M['logz'][-1])
    postsamples = dynesty.utils.resample_equal(M.samples, weights)

    print('\n Number of posterior samples is {}'.format(postsamples.shape[0]))


    return M
