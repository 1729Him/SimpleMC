[docs]    def RHSquared_a(self, a):
        NuContrib = self.NuDensity.rho(a) / self.h ** 2
        rhow = a ** (-3 * (1.0 + self.w0 + self.wa)) * N.exp(-3 * self.wa * (1 - a))
        return (self.Ocb / a ** 3 + self.Ok / a ** 2 + self.Omrad / a ** 4 + NuContrib + (
                    1.0 - self.Om - self.Ok) * rhow)
